                            Thanks, my solution for the task # 1:

                            'use strict';

                            populate();

                            document.addEventListener('scroll', function () {
                                populate();
                            });

                            function populate( {
                            	target = document.scrollingElement,
                            	minDistanceToBottom = 50,
                            } = {}
                            ) {
                                let leftToBottom = getDistanceToBottom(target);

                                while (leftToBottom < minDistanceToBottom) {
                            	let currentDate = document.createElement('div');

                            	currentDate.innerHTML = new Date();
                            	target.append(currentDate);

                            	leftToBottom = getDistanceToBottom(target);
                                }
                            }

                            function getDistanceToBottom(target) {
                                return target.scrollHeight - target.clientHeight - target.scrollTop;
                            }
